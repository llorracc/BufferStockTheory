% PhaseB1: \BSTcondref is the conditional fallback for the named-condition
% macros formerly written as \hyperlink{X}{\textrm{Y}}. KaTeX rejects
% \hyperlink in math mode, so the MyST build path swaps in \textsf{#2}.
% When Web=false (default for pdflatex), the expansion is identical to the
% original \hyperlink{#1}{\textrm{#2}}, so PDF output is byte-identical.
\providecommand{\BSTcondref}[2]{%
  \ifthenelse{\boolean{Web}}{\textsf{#2}}{\hyperlink{#1}{\textrm{#2}}}%
}

% PhaseB2: alias rewrites for \ifthenelse{\boolean{Web}}{...}{...}.
% Defined here in the preamble so they are available before the .sty
% load (the master itself uses \BothVersions / \PDFOnly below). The .sty
% (loaded later) carries idempotent \providecommand copies so body
% subfiles compiled standalone still see them.
\providecommand{\PDFOnly}[1]{\ifthenelse{\boolean{Web}}{}{#1}}
\providecommand{\WebOnly}[1]{\ifthenelse{\boolean{Web}}{#1}{}}
\providecommand{\BothVersions}[2]{\ifthenelse{\boolean{Web}}{#1}{#2}}
